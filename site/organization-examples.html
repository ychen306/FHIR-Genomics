5.4.6.2 Gastro

XML

Gastroenterology team at ACME Healthcare (id = "1")

Raw XML

<Organization xmlns="http://hl7.org/fhir">
  <text>
    <status value="generated"/>
    <div xmlns="http://www.w3.org/1999/xhtml">
      <p>Gastroenterology @ Acme Hospital. ph: +1 555 234 3523, email: <a href="mailto:gastro@acme.org">gastro@acme.org</a></p>
    </div>
  </text>

  <!--   Clinical Team &quot;Gastroenterology&quot; at Acme Hospital   -->
  <identifier>
    <system value="http://www.acme.org.au/units"/>
    <value value="Gastro"/>
  </identifier>
  <name value="Gastroenterology"/>
  <telecom>
    <system value="phone"/>
    <value value="+1 555 234 3523"/>
    <use value="mobile"/>
  </telecom>
  <telecom>
    <system value="email"/>
    <value value="gastro@acme.org"/>
    <use value="work"/>
  </telecom>
  
  <partOf>
    <reference value="Organization/1"/>
    <display value="ACME Healthcare, Inc"/>
  </partOf>
</Organization>
